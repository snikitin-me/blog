Работа со строками

JS

// string literal
"don't say \"no\""
'don\'t say "no"'


// variable interpolation
1 + "1"
"1" + 1


// string concatenate
s = "Hello, " + "World!";


// string replicate
var hbar = Array(80).join("-");


// string join
["do", "re", "mi"].join(" ")


// Split
// [ "do", "re", "", "mi", "" ]:
"do re  mi ".split(" ")
// [ "do", "re", "mi", "" ]:
"do re  mi ".split(/\s+/)


// format string
// None; use string concatenation.
// Evaluates to "12.35":
12.3456.toFixed(2)


// translate case to upper, to lower
"lorem".toUpperCase()
"LOREM".toLowerCase()


// trim
" lorem ".trim()
// some browsers:
" lorem".trimLeft()
"lorem ".trimRight()


// string to number
7 + parseInt("12", 10)
73.9 + parseFloat(".037")
// 12:
parseInt("12A")
// NaN:
parseInt("A")


// number to string
"value: " + 8


// prefix and suffix test
"foobar".startsWith("foo")
"foobar".endsWith("bar")


// length
"lorem".length  // 5


// index of substring
// returns -1 if not found:
"lorem ipsum".indexOf("ipsum")
["lorem","ipsum"].indexOf("ipsum")


// extract substring
"lorem ipsum".substr(6, 5)
"lorem ipsum".substring(6, 11)


// character lookup
"lorem ipsum"[6]


// chr and ord
String.fromCharCode(65)
"A".charCodeAt(0)


// to array of characters
"abcd".split("")

Работа с датами

JS

var t = new Date();


// to unix epoch, from unix epoch
Math.round(t.getTime() / 1000)


var epoch = 1315716177;
var t2 = new Date(epoch * 1000);


// parse date w/o format
var t = new Date("July 7, 1999");


// get date parts
t.getFullYear()
t.getMonth() + 1
t.getDate() // getDay() is day of week


// get time parts
t.getHours()
t.getMinutes()
t.getSeconds()


// build date/time from parts
var yr = 1999;
var mo = 9;
var dy = 10;
var hr = 23;
var mi = 30;
var ss = 0;
var t = new Date(yr, mo - 1, dy, hr, mi, ss);


// add time duration
var t1 = new Date();
var delta = (10 * 60 + 3) * 1000;
var t2 = new Date(t1.getTime() + delta);
